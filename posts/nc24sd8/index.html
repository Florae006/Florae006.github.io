void solve() {
    int n;cin >> n;
    vector<int>a(n);
    vector<bool>vis(100005, false);
    int mx = -1;
    for (int i = 0;i < n;i++) {
        cin >> a[i];
        vis[a[i]] = true;
        if (mx != -1)mx = max(mx, a[i]);
        else mx = a[i];
    }
    if (n == 1) {
        cout << "Haitang\n";return;
    }
    int ans = 0;
    for (int i = 1;i <= mx;i++) {
        if (vis[i])continue;
        vector<int>g;
        for (int j = 2 * i;j <= mx;j += i) {
            if (vis[j])g.push_back(j);
        }
        if (g.size() >= 2) {
            int k = gcd(g[0], g[1]);
            for (int j = 2;j < g.size();j++) {
                k = gcd(k, g[j]);
            }
            if (k == i) {
                ans++;
                vis[i] = true;
            }
        }
    }
    if (ans % 2) {
        cout << "dXqwq\n";
    }
    else {
        cout << "Haitang\n";
    }
}
