ll maxsum(vector<ll>a) {
    int n = a.size();
    ll sum = a[0];

        if (b > sum)sum = b;
    }
    return sum;
}

void solve() {
    int n;cin >> n;
    vector<ll>a(n);
    map<int, int>cnt;
    for (int i = 0;i < n;i++) {
        cin >> a[i];
        cnt[a[i] % 36]++;
    }
    ll k = 1;
    vector<int>dk(19);
    for (int i = 1;i < 19;i++) {
        k *= 10;
        dk[i] = k % 36;
    }
    auto getlen = [](ll x) {
        int i = 0;
        while (x) {
            i++;x /= 10;
        }
        return i;
        };
    ll ans = 0;
    for (int i = 0;i < n;i++) {
        int x = a[i] % 36;
        int z = getlen(a[i]);
        for (int j = 0;j < 36;j++) {
            if ((j * dk[z] + x) % 36 == 0) {
                ans += cnt[j];
                if (j == x)ans -= 1;
            }
        }
    }
    cout << ans << '\n';
}
