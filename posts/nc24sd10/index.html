#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn = 1050;
struct point {
    int x, y;
};
int X[maxn], Y[maxn];
bool vis[maxn][maxn];
void solve() {
    int n;cin >> n;
    vector<point>v;
    for (int i = 1;i <= n * n;i++) {
        int x1, y1;cin >> x1 >> y1;
        if (X[x1] + 1 > 2 || Y[y1] + 1 > 2 || vis[x1][y1]) {
            cout << "0";
            continue;
        }
        X[x1]++, Y[y1]++;
        if (i == 1) {
            v.push_back({ x1,y1 });
            vis[x1][y1] = true;
            cout << "1";continue;
        }
        for (auto [x2, y2] : v) {
            if (y2 == y1 || x2 == x1)continue;
            int dx = abs(x2 - x1), dy = abs(y2 - y1), c = (1.0 * (y1 - y2) / (x1 - x2) > 0 ? 1 : -1);
            int g = gcd(dx, dy);
            dx /= g, dy /= g;
            int x = x1, y = y1;
            while (x >= 1 && x <= n && y >= 1 && y <= n) {
                vis[x][y] = true;
                x += dx, y += c * dy;
            }
            x = x1, y = y1;
            while (x >= 1 && x <= n && y >= 1 && y <= n) {
                vis[x][y] = true;
                x -= dx, y -= c * dy;
            }
        }
        v.push_back({ x1,y1 });
        cout << "1";
    }
    cout << "\n";
}

int main() {
    int t = 1;
    // cin >> t;cin.get();
    while (t--)
        solve();

    return 0;
}
