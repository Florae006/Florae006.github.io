    vector<ll>a(n);

    for (int i = 0;i < n;i++) {

    for (ll k = 1;k * k <= sum;k++) {
        if (sum % k == 0) {
            ll x = sum / k;
            if (x >= n && k <= f) {
                res.insert(k);

                res.insert(sum / k);

        }
