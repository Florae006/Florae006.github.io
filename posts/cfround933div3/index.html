void solve() {
    ll n, m, k, d;
    cin >> n >> m >> k >> d;
    vector<vector<ll>>a(n, vector<ll>(m));
    for (ll i = 0;i < n;i++) {
        for (ll j = 0;j < m;j++) {
            cin >> a[i][j];
            a[i][j] += 1;
        }
    }
    // æ¯è¡Œdp
    vector<ll>spd(n, 0);
    for (ll ni = 0;ni < n;ni++) {
        vector<ll>dp(m, 0);
        for (ll i = 0;i < m;i++) {
            dp[i] = a[ni][i];
        }
        // ç»´æŠ¤å‰dä¸ªå€¼ä¸­çš„æœ€å°å€¼
        queue<ll>q1;
        deque<ll>qmin;
        q1.push(dp[0]);
        qmin.push_back(dp[0]);

        for (ll i = 1;i < m;i++) {
            if (i == 1)
                dp[i] += dp[0];
            else {
                ll mn = qmin.front();
                dp[i] = mn + dp[i];
            }

            if (q1.size() == d + 1) {
                if (q1.front() == qmin.front()) {
                    qmin.pop_front();
                }
                q1.pop();
            }
            q1.push(dp[i]);
            while (!qmin.empty() && dp[i] < qmin.back()) {
                qmin.pop_back();
            }
            qmin.push_back(dp[i]);
        }
        spd[ni] = dp[m - 1];
    }

    vector<ll>sum(n + 1, 0);
    sum[0] = 0;
    for (ll i = 1;i <= n;i++) {
        sum[i] = sum[i - 1] + spd[i - 1];
    }
    ll ans = sum[k];
    for (ll i = k;i <= n;i++) {
        ans = min(ans, sum[i] - sum[i - k]);
    }
    cout << ans << endl;
}
