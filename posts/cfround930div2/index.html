    cin >> s[0] >> s[1];
    string st = s[0] + s[1];
    ll cnt = 1;
    ll ix = 2;
    for (int i = 2;i + n - 2 < 2 * n - 1;i++) {
        ll j = i + n - 2;//[i,j]
        if (st[i - 1] == '0' && st[j] == '1') {
            cnt = 1;
            ix = i;
        }
        else if (st[i - 1] == st[j]) {
            cnt++;
        }
        else {
            break;
        }
    }

    string res = st.substr(0, ix) + st.substr(ix + n - 1);
    string res1 = s[0][0] + s[1];
    if (res1 < res)res = res1;

    cout << res << '\n';
    cout << cnt << '\n';
}
