        ck[x] = true;

    bool f = false;
    int y = -1;

            y = i;

        cout << -1 << endl;
        return;

        }
        return;

    int p = 1;
    int cnt = 0;
    vector<pair<int, int>>ans;
    vector<bool>hs(y, false);
    vector<bool>hsf(y, false);
    for (int i = 1;i <= n;i++) {
        if (a[i] < y && !hs[a[i]]) {
            hs[a[i]] = true;
            cnt++;
        }
        if (cnt == y) {
            ans.push_back({ p, i });
            p = i + 1;
            cnt = 0;
            // ä¼šä¸ä¼šTle
            hs = hsf;
        }

    if (ans.size() == 1) {

    for (int i = 0;i < ans.size();i++) {
        if (i != ans.size() - 1)
            cout << ans[i].first << " " << ans[i].second << endl;
        else {
            cout << ans[i].first << " " << n << endl;
        }
    }

}
