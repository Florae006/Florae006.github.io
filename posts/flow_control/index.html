        clearTimeout(timeout);

            fn.apply(context, args);
        }, wait);
    }
}

class RequestQueue {
  constructor(maxConcurrent) {
    this.maxConcurrent = maxConcurrent; // æœ€å¤§å¹¶å‘æ•°
    this.queue = [];  // è¯·æ±‚é˜Ÿåˆ—
    this.currentlyRunning = 0;  // å½“å‰æ­£åœ¨è¿è¡Œçš„è¯·æ±‚æ•°
  }

  add(request) {
    return new Promise((resolve, reject) => {
      this.queue.push({ request, resolve, reject });
      this.processQueue();
    });
  };

  processQueue() {
    if (this.queue.length > 0 && this.currentlyRunning < this.maxConcurrent) {
      const { request, resolve, reject } = this.queue.shift();
      this.currentlyRunning++;
      request().then(resolve).catch(reject).finally(() => {
        this.currentlyRunning--;
        this.processQueue();
      });
    };
  };
}

class RateLimiter {
  constructor(limit, interval) {
    this.limit = limit;
    this.interval = interval;
    this.queue = [];
    this.times = [];
  }

  add(request) {
    this.queue.push(request);
    this.run();
  }

  run() {
    const now = Date.now();
    this.times = this.times.filter(time => now - time < this.interval);

    if (this.times.length < this.limit && this.queue.length) {
      const request = this.queue.shift();
      this.times.push(now);
      request();
    }

    if (this.queue.length) {
      setTimeout(() => this.run(), this.interval - (now - this.times[0]));
    }
  }
}
